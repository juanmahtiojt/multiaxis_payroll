    <script>
        function searchTable() {
            const input = document.getElementById('searchName');
            const filter = input.value.toLowerCase();
            const table = document.getElementById('employeeTable');
            const trs = table.getElementsByTagName('tr');

            for (let i = 1; i < trs.length; i++) {
                const tds = trs[i].getElementsByTagName('td');
                let match = false;
                for (let j = 0; j < tds.length; j++) {
                    if (tds[j]) {
                        const tdText = tds[j].textContent || tds[j].innerText;
                        if (tdText.toLowerCase().indexOf(filter) > -1) {
                            match = true;
                            break;
                        }
                    }
                }
                trs[i].style.display = match ? '' : 'none';
            }
        }

        function clearSearch() {
            document.getElementById('searchName').value = '';
            searchTable();
        }

        function toggleNameGroup(name) {
            const group = document.getElementById(name);
            const button = document.querySelector(`button[onclick='toggleNameGroup("${name}")']`);

            if (group.style.display === 'none') {
                group.style.display = '';
                button.textContent = 'Minimize'; // Change to 'Minimize'
            } else {
                group.style.display = 'none';
                button.textContent = 'Expand'; // Change to 'Expand'
            }
        }
    </script>
