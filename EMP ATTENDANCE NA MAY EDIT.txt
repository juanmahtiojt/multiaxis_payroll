<script>
function enableEdit(button) {
    const password = prompt("Enter password to edit:");
    if (password !== "admin123") {
        alert("Incorrect password!");
        return;
    }

    const row = button.closest("tr");
    const amInCell = row.querySelector(".am-in");
    const amOutCell = row.querySelector(".am-out");

    const amInText = amInCell.textContent.trim();
    const amOutText = amOutCell.textContent.trim();

    amInCell.innerHTML = `<input type="text" class="form-control" value="${amInText}">`;
    amOutCell.innerHTML = `<input type="text" class="form-control" value="${amOutText}">`;

    amInCell.classList.add("editable");
    amOutCell.classList.add("editable");

    button.classList.add("d-none");
    row.querySelector("button.btn-success").classList.remove("d-none");
}

function saveRow(button) {
    const row = button.closest("tr");
    const id_no = row.children[0].textContent.trim();
    const department = row.children[1].textContent.trim();
    const name = row.children[2].textContent.trim();
    const date = row.children[3].textContent.trim();

    const amInInput = row.querySelector(".am-in input");
    const amOutInput = row.querySelector(".am-out input");

    const am_in = amInInput.value;
    const am_out = amOutInput.value;

    fetch('save_attendance.php', {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: `id_no=${encodeURIComponent(id_no)}&department=${encodeURIComponent(department)}&name=${encodeURIComponent(name)}&date=${encodeURIComponent(date)}&am_in=${encodeURIComponent(am_in)}&am_out=${encodeURIComponent(am_out)}`
    })
    .then(res => res.text())
    .then(response => {
        alert(response);
        row.querySelector(".am-in").innerHTML = am_in;
        row.querySelector(".am-out").innerHTML = am_out;

        row.querySelector(".am-in").classList.remove("editable");
        row.querySelector(".am-out").classList.remove("editable");

        button.classList.add("d-none");
        row.querySelector("button.btn-primary").classList.remove("d-none");
    })
    .catch(err => {
        console.error(err);
        alert("Error saving attendance.");
    });
}
</script>
